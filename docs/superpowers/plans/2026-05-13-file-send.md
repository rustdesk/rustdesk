# File Send Feature Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a 📎 button to PowerStrip row 2 that opens a bottom sheet for sending one or more files from iPhone to the remote machine.

**Architecture:** A new `FileSendSheet` StatefulWidget handles all three states (destination picker → sending → done). `PowerStrip` gets a new `onFileSend` callback wired from `RemoteSessionScreen._onFileSend()`. Progress is tracked by reading `gFFI.fileModel.jobController.jobTable` (an `RxList`) via an `Obx` wrapper inside the sheet.

**Tech Stack:** Flutter/Dart, `file_picker: ^5.1.0` (already in pubspec), RustDesk `bind.sessionSendFiles` FFI, `get` (GetX for `Obx`/`.obs`)

import 'dart:io';

import 'package:file_picker/file_picker.dart';
import 'package:flutter/material.dart';
import 'package:flutter_hbb/common.dart';
import 'package:flutter_hbb/models/file_model.dart';
import 'package:get/get.dart';
import 'package:path/path.dart' as p;

enum _SheetState { picking, sending, done }

class FileSendSheet extends StatefulWidget {
  final FFI ffi;
  const FileSendSheet({super.key, required this.ffi});

  @override
  State<FileSendSheet> createState() => _FileSendSheetState();
}

class _FileSendSheetState extends State<FileSendSheet> {
  _SheetState _state = _SheetState.picking;

  // destination picker
  final _pathController = TextEditingController(text: '~');
  String? _selectedChip = '~';
  static const _commonPaths = [
    ('🏠', '~', 'Home'),
    ('🖥', '~/Desktop', 'Desktop'),
    ('⬇️', '~/Downloads', 'Downloads'),
    ('📄', '~/Documents', 'Documents'),
    ('🗂', '/tmp', '/tmp'),
  ];

  // transfer state
  List<File> _files = [];
  bool _accordionExpanded = false;

  // job IDs tracked per file (index matches _files)
  final List<int> _jobIds = [];

  String get _destination => _pathController.text.trim();

  @override
  void dispose() {
    _pathController.dispose();
    super.dispose();
  }

  Future<void> _pickAndSend() async {
    final result = await FilePicker.platform.pickFiles(allowMultiple: true);
    if (result == null || result.files.isEmpty) return;
    final files = result.files
        .where((f) => f.path != null)
        .map((f) => File(f.path!))
        .toList();
    if (files.isEmpty) return;
    await _startTransfer(files);
  }

  Future<void> _startTransfer(List<File> files) async {
    _files = files;
    _jobIds.clear();
    _accordionExpanded = false;
    setState(() => _state = _SheetState.sending);

    // Resolve ~ to absolute home directory
    String dest = _destination;
    if (dest.startsWith('~')) {
      final home = await bind.mainGetHomeDir();
      dest = dest.replaceFirst('~', home);
    }

    final fileModel = widget.ffi.fileModel;

    for (final file in files) {
      final jobId = fileModel.jobController.addTransferJob(
        Entry()
          ..path = file.path
          ..name = p.basename(file.path)
          ..size = await file.length()
          ..isFile = true,
        false, // isRemoteToLocal = false (local → remote)
      );
      _jobIds.add(jobId);
      bind.sessionSendFiles(
        sessionId: widget.ffi.sessionId,
        actId: jobId,
        path: file.path,
        to: dest,
        fileNum: 0,
        includeHidden: false,
        isRemote: false,
        isDir: false,
      );
    }
  }

  void _reset() {
    _files.clear();
    _jobIds.clear();
    _pathController.text = '~';
    _selectedChip = '~';
    setState(() => _state = _SheetState.picking);
  }

  @override
  Widget build(BuildContext context) {
    return SafeArea(
      child: switch (_state) {
        _SheetState.picking => _buildPicker(),
        _SheetState.sending => _buildSending(),
        _SheetState.done => _buildDone(),
      },
    );
  }

  // ── Sending state ─────────────────────────────────────────────────────────

  Widget _buildSending() {
    return Obx(() {
      final jobs = widget.ffi.fileModel.jobController.jobTable;
      final myJobs = _jobIds
          .map((id) {
            final idx = jobs.indexWhere((j) => j.id == id);
            return idx >= 0 ? jobs[idx] : null;
          })
          .whereType<JobProgress>()
          .toList();

      final allDone = myJobs.isNotEmpty && myJobs.every((j) => j.state == JobState.done || j.state == JobState.error);
      if (allDone) {
        WidgetsBinding.instance.addPostFrameCallback((_) {
          if (mounted) setState(() => _state = _SheetState.done);
        });
      }

      final totalSize = myJobs.fold<int>(0, (s, j) => s + j.totalSize);
      final finishedSize = myJobs.fold<int>(0, (s, j) => s + j.finishedSize);
      final pct = totalSize > 0 ? (finishedSize / totalSize).clamp(0.0, 1.0) : 0.0;
      final speed = myJobs.fold<double>(0, (s, j) => s + j.speed);

      return Column(
        mainAxisSize: MainAxisSize.min,
        crossAxisAlignment: CrossAxisAlignment.stretch,
        children: [
          _handle(),
          _sheetTitle('Sending ${_files.length == 1 ? 'File' : 'Files'}',
              'To ${_destination}'),
          const Divider(height: 1, color: Color(0xFF2C2C2E)),
          const SizedBox(height: 14),
          if (_files.length == 1)
            _singleFileProgress(myJobs.isNotEmpty ? myJobs.first : null, _files.first, pct, speed)
          else
            _multiFileAccordion(myJobs, pct, speed, totalSize, finishedSize),
          const Spacer(),
          Padding(
            padding: const EdgeInsets.fromLTRB(20, 8, 20, 28),
            child: _secondaryButton(label: 'Cancel', destructive: true, onTap: _cancelAll),
          ),
        ],
      );
    });
  }

  Widget _fileRow(File file, JobProgress? job) {
    final state = job?.state ?? JobState.none;
    final pct = (job != null && job.totalSize > 0)
        ? (job.finishedSize / job.totalSize).clamp(0.0, 1.0)
        : 0.0;

    Widget badge;
    Color barColor;
    if (state == JobState.done) {
      badge = const Text('✓ Done', style: TextStyle(fontSize: 12, color: Color(0xFF30D158), fontWeight: FontWeight.w500));
      barColor = const Color(0xFF30D158);
    } else if (state == JobState.error) {
      badge = const Text('✗ Error', style: TextStyle(fontSize: 12, color: Color(0xFFFF453A), fontWeight: FontWeight.w500));
      barColor = const Color(0xFFFF453A);
    } else if (state == JobState.inProgress) {
      badge = Text('${(pct * 100).toStringAsFixed(0)}%',
          style: const TextStyle(fontSize: 12, color: Color(0xFF0A84FF), fontWeight: FontWeight.w500));
      barColor = const Color(0xFF0A84FF);
    } else {
      badge = const Text('Queued', style: TextStyle(fontSize: 12, color: Color(0xFF636366)));
      barColor = const Color(0xFF0A84FF);
    }

    return Padding(
      padding: const EdgeInsets.only(bottom: 10),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          Row(children: [
            Text(_fileIcon(file.path), style: const TextStyle(fontSize: 14)),
            const SizedBox(width: 8),
            Expanded(
              child: Text(p.basename(file.path),
                  maxLines: 1,
                  overflow: TextOverflow.ellipsis,
                  style: TextStyle(
                      fontSize: 13,
                      color: state == JobState.none
                          ? const Color(0xFF636366)
                          : const Color(0xFFEBEBF5))),
            ),
            badge,
          ]),
          const SizedBox(height: 4),
          ClipRRect(
            borderRadius: BorderRadius.circular(2),
            child: LinearProgressIndicator(
              value: pct,
              minHeight: 4,
              backgroundColor: const Color(0xFF3A3A3C),
              valueColor: AlwaysStoppedAnimation(barColor),
            ),
          ),
        ],
      ),
    );
  }

  // ── Done state ────────────────────────────────────────────────────────────
